## Devcontainer using the pyment-public dependencies connected to ver 3.0.1 from multitask branch
FROM mcr.microsoft.com/devcontainers/python:3.9.23

RUN apt-get update && apt-get install -y git

# Clone external repo at a specific version/tag
RUN git clone --branch multitask https://github.com/estenhl/pyment-public.git /opt/external-repo
	&& cd /opt/external-repo\
	&& git checkout 3.0.1
RUN pip install poetry


RUN cd /opt/external-repo && poetry install
